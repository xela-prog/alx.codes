        if (e.isIntersecting) e.target.classList.add('visible');
      });
    }, { threshold: 0.12 });
    reveals.forEach(el => io.observe(el));
  }
})();

# alx.codes Portfolio

A polished Everforest/OpenCode-inspired full-stack developer portfolio for Pierre-Alexandre Merge Fignole.

## Run locally

```bash
npm install
npm run dev
```

Open `index.html` with Live Server or your preferred local server.

## Build CSS

```bash
npm run build
```

## Customize

- Replace contact email placeholders in `index.html`.
- Replace social links marked with `href="#"`.
- Replace project screenshots in `assets/images/projects/`:
  - `finance-terminal.png`
  - `openpad.png`
  - `client-redesign.png`
- Replace `assets/images/profile.png` with your profile image.
- Update project names, descriptions, tech stacks, and links in the `#work` section.

## Design system

Palette is based on Everforest/OpenCode:

- Background: `#2D353B`
- Surface: `#343F44`
- Elevated: `#3D484D`
- Text: `#D3C6AA`
- Muted: `#9DA9A0`
- Green: `#A7C080`
- Aqua: `#83C092`
- Yellow: `#DBBC7F`
- Orange: `#E69875`
- Red: `#E67E80` only for warning/error accents

        completion.choices?.[0]?.message?.content?.trim() ||
        "The assistant could not generate a response."

    });

  } catch (error) {

    console.error("\nGROQ ERROR:\n");
    console.error(error);

    return res.status(500).json({

      reply:
        "The assistant is temporarily unavailable."

    });

  }

});

  res.sendFile(
    path.join(__dirname, "index.html")
  );

});

  console.log(
    `\nServer running at http://localhost:${PORT}\n`
  );
